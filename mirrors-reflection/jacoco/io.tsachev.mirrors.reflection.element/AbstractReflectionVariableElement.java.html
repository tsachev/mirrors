package io.tsachev.mirrors.reflection.element;

import java.lang.reflect.AnnotatedElement;
import javax.lang.model.element.ElementVisitor;
import javax.lang.model.type.TypeMirror;

/**
 * @author Vladimir Tsanev
 */
public abstract class AbstractReflectionVariableElement<S extends AnnotatedElement>
    extends AbstractReflectionElement<S> implements ReflectionVariableElement {

  protected AbstractReflectionVariableElement(S source) {
    super(source);
  }

  @Override
  public TypeMirror asType() {
    return null; //// TODO: 7/25/16  
  }

  @Override
  public <R, P> R accept(ElementVisitor<R, P> visitor, P param) {
    return visitor.visitVariable(this, param);
  }

}
