package io.tsachev.mirrors.tck.playground;

public class NestingClass {
  static final int answer;

  static {
    @MyAnnotation("classInStaticInitializer")
    class ClassInStaticInitializer {
      private String field;
    }

    answer = 42;
  }

  {
    @MyAnnotation("classInInstanceInitializer")
    class ClassInInstanceInitializer {
      private String field;
    }
  }

  NestingClass() {

    @MyAnnotation("classInConstructor")
    class ClassInConstructor {
      private String field;
    }
  }

  @MyAnnotation("staticNestedClass")
  static class StaticNestedClass {

  }

  @MyAnnotation("instanceNestedClass")
  class InstanceNestedClass {
    private String field;
  }

  void foo() {
    @MyAnnotation("classInMethod")
    class ClassInMethod {
      private String field;
    }
  }
}
