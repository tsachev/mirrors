package io.tsachev.mirrors.reflection.type;

import javax.lang.model.type.NullType;
import javax.lang.model.type.TypeKind;
import javax.lang.model.type.TypeVisitor;

/**
 * @author Vladimir Tsanev
 */
public class ReflectionNullType extends AbstractReflectionTypeMirror implements NullType {

  private static final ReflectionNullType instance = new ReflectionNullType();

  private ReflectionNullType() {
    super(TypeKind.NULL);
  }

  @Override
  public <R, P> R accept(TypeVisitor<R, P> visitor, P param) {
    return visitor.visitNull(this, param);
  }

  public static ReflectionNullType getInstance() {
    return instance;
  }
}
