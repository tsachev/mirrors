package io.tsachev.mirrors.reflection.type;

import javax.lang.model.type.ArrayType;
import javax.lang.model.type.TypeKind;
import javax.lang.model.type.TypeVisitor;

/**
 * @author Vladimir Tsanev
 */
public class ReflectionArrayType extends AbstractReflectionTypeMirror implements ArrayType {

  private final Class<?> source;

  static ReflectionArrayType newInstance(Class<?> source) {
    if (!source.isArray()) {
      throw new IllegalArgumentException();
    }
    return new ReflectionArrayType(source);
  }

  private ReflectionArrayType(Class<?> source) {
    super(TypeKind.ARRAY);
    this.source = source;
  }

  @Override
  public ReflectionTypeMirror getComponentType() {
    return null;
  }

  @Override
  public <R, P> R accept(TypeVisitor<R, P> visitor, P param) {
    return visitor.visitArray(this, param);
  }
}
