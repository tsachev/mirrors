125   @Override
126   public ReflectionPrimitiveType getPrimitiveType(TypeKind kind) {
127     switch (kind) {
128       case BOOLEAN:
129         return ReflectionPrimitiveType.getBooleanInstance();
130       case BYTE:
131         return ReflectionPrimitiveType.getByteInstance();
132       case SHORT:
133         return ReflectionPrimitiveType.getShortInstance();
134       case INT:
135         return ReflectionPrimitiveType.getIntInstance();
136       case LONG:
137         return ReflectionPrimitiveType.getLongInstance();
138       case CHAR:
139         return ReflectionPrimitiveType.getCharInstance();
140       case FLOAT:
141         return ReflectionPrimitiveType.getFloatInstance();
142       case DOUBLE:
143         return ReflectionPrimitiveType.getDoubleInstance();
144       default:
145         throw new IllegalArgumentException("Kind must be primitive kind, but is: " + kind);
146     }
